#include <stdlib.h>
#include <stdio.h>
#include <Windows.h>
#define SIZE 100
int main()
{
        SetConsoleCP(1251);
        SetConsoleOutputCP(1251);

        FILE *mf;
        char list[SIZE], word;
        int  amount = 0;

        printf("Введіть текс : ");
        fgets(list, SIZE, stdin);
        mf = fopen("mf.txt", "w");
        if (mf == NULL) {
            perror("Файл не знайдено");
            return(1);
        }
        fclose(mf);
	mf = fopen("mf.txt", "r");
        fgets(list, SIZE, mf);
        printf("Введіть символ який шукаємо : ");
        scanf("%c", &word);
        for (int i = 0; i < SIZE && list[i] != '\0'; i++)
        {
            if (list[i] == word)
                amount++;
        }
        printf("Символів %c в тексті: %d", word, amount);
        fclose(mf);
        return 0;
}
